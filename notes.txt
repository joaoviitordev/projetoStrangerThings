});
 
# ScrollSmoother => Scroll Suave:

ScrollSmoother.create ({
    smooth: 1.5,   ==> indica o nível de suavidade do Scroll
    effects: true  ==> ativa as animações para uma determinada coisa do HTML, atraves do parâmetro data-speed=""
});

# animações de Movimento:

gsap.from("elemento que quero a animações", {  ==> gsap.from indica de onde o elemento vem 
    ... qual animação quero fazer ...
})

gsap.to("elemento que quero a animações", {    ==> gsap.from indica para onde o elemento vai
    ... qual animação quero fazer ...
})

# ScrollTrigger ==> Ele anima os elementos assim que a barra de rolagem chega em um determinado local da página
